\def\connect#1#2#3#4{
  \draw [draw=#3, fill=#3] (#1.south west) -- ($(#1.south)!0.5!(#2.north)$) -- (#1.south east) -- cycle;
  \fill [fill=#4] (#1.south west) -- (#2.north west) -- (#2.north east) -- (#1.south east) -- ($(#1.south)!0.5!(#2.north)$) -- cycle;
}
